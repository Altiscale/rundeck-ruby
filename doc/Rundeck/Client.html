def objectify(result)
  if result.is_a?(Hash)
    ObjectifiedHash.new(result)
  elsif result.is_a?(Array)
    result.map { |e| ObjectifiedHash.new(e) }
  elsif result.nil?
    nil
  else
    fail Error::Parsing, "Couldn't parse a response body"
  end
end
