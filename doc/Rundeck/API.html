def initialize(options = {})
  options = Rundeck.options.merge(options)
  Configuration::VALID_OPTIONS_KEYS.each do |key|
    send("#{key}=", options[key]) if options.key?(key)
  end
  set_request_defaults @endpoint, @api_token
end
